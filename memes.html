    function filterCategory(cat) {

        document.querySelectorAll('.tab').forEach(t => t.classList.remove('tab-active'));

        document.getElementById('tab-' + cat).classList.add('tab-active');


        document.querySelectorAll('#meme-grid > div').forEach(card => {

            if (cat === 'all') card.style.display = 'block';

            else card.style.display = card.getAttribute('data-category') === cat ? 'block' : 'none';

        });

    }


    function openMemeModal(id) {

        const data = memeData[id];

        if (!data) return;

        currentMemeId = id;


        document.getElementById('modal-image').innerHTML = `<img src="${data.image}" class="w-full h-auto max-h-[70vh] md:max-h-[65vh] object-contain mx-auto" alt="${data.title}">`;

        document.getElementById('context-panel').classList.add('hidden');

        document.getElementById('modal-buttons').classList.remove('hidden');


        const modal = document.getElementById('meme-modal');

        modal.classList.remove('hidden');

        document.body.style.overflow = 'hidden';


        document.addEventListener('keydown', handleMemeEscape);

    }


    function closeMemeModal() {

        const modal = document.getElementById('meme-modal');

        modal.classList.add('hidden');

        document.body.style.overflow = 'visible';

        document.removeEventListener('keydown', handleMemeEscape);

    }


    function handleMemeEscape(e) {

        if (e.key === 'Escape') closeMemeModal();

    }


    document.getElementById('meme-modal').addEventListener('click', function(e) {
